310template<IsACComponentChild T, typename ...Args>
311CWeakPtr<T> CGameObject::AddComponent(Args&&... args) {
312 // Add component to array
313 m_components.emplace_back(CUniquePtrWeakable<T>::Make(this, std::forward<Args>(args)...));
314
315 // Call awake processing
316 CWeakPtr<T> addedComponent = m_components.back().GetWeakPtr();
317 addedComponent->Awake();
318 addedComponent->OnEnable();
319
320 // If T is derived class of ACCollider3D, Add it to the array of colliders
321 if constexpr (std::is_base_of_v<ACCollider3D, T>) {
322 ColliderAdded(addedComponent);
323 }
324
325 // Return a weak pointer to the added component
326 return addedComponent;
327}
