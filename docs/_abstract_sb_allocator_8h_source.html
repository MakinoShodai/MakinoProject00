132// Copy data directly
133template <class StructBuffType>
134Utl::Dx::CPU_DESCRIPTOR_HANDLE ACSbAllocater<StructBuffType>::DirectDataCopy(MeshKey key, const StructBuffType* src) {
135 // Get next back buffer index
136 UINT bufferIndex = CSwapChain::GetMain().GetNextBackBufferIndex();
137
138 // If mesh data doesn't exist in the map, return null data
139 auto it = m_dataPerMesh.find(key);
140 if (it == m_dataPerMesh.end()) {
141 throw Utl::Error::CStopDrawingSceneError(L"Warning! Structured buffer data for a sent mesh doesn't exist map!\n");
142 }
143
144 // Perform map operation
145 m_buffers[*it->second.handle].structuredBuffers[bufferIndex].Map();
146
147 // Update data of the structured buffer
148 m_buffers[*it->second.handle].structuredBuffers[bufferIndex].DataCopy(src, it->second.elementNum, it->second.offset);
149
150 // Update last updated frame number
151 it->second.lastUpdateFrame = m_currentFrameNumber;
152
153 // Get specified CPU handle from descriptor heap
154 return m_buffers[*it->second.handle].descriptorHeaps[bufferIndex].GetCpuHandle(it->second.meshIndex);
155}

158template <class StructBuffType>
159void ACSbAllocater<StructBuffType>::CreatePerComponentBuffer(UINT size, UINT viewNum) {
160 m_buffers.EmplaceBack();
161 auto& newBuffer = m_buffers[*m_buffers.BackHandle()];
162 // Create structured buffers and descriptor heaps
163 for (UINT i = 0; i < SCREEN_BUFFERING_NUM; ++i) {
164 newBuffer.structuredBuffers[i].Create(size);
165 newBuffer.descriptorHeaps[i].Create(D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV, D3D12_DESCRIPTOR_HEAP_FLAG_NONE, viewNum);
166 }
167}

169// Create a per mesh data
170template <class StructBuffType>
171void ACSbAllocater<StructBuffType>::CreatePerMeshData(UINT meshIndex, MeshKey meshKey, UINT elementNum, UINT offset) {
172 // Emplace data in the map
173 WeakStableHandle<SbDataPerComponent<StructBuffType>> handle = m_buffers.BackHandle();
174 m_dataPerMesh.emplace(meshKey, SbDataPerMesh<StructBuffType>());
175
176 // Copy needed data
177 SbDataPerMesh<StructBuffType>& newData = m_dataPerMesh[meshKey];
178 newData.meshIndex = meshIndex;
179 newData.handle = handle;
180 newData.elementNum = elementNum;
181 newData.offset = offset;
182 newData.lastUpdateFrame = m_currentFrameNumber - 1;
183
184 // Create shader resource view desc
185 D3D12_SHADER_RESOURCE_VIEW_DESC srvDesc = {};
186 srvDesc.Format = DXGI_FORMAT_UNKNOWN;
187 srvDesc.ViewDimension = D3D12_SRV_DIMENSION_BUFFER;
188 srvDesc.Shader4ComponentMapping = D3D12_DEFAULT_SHADER_4_COMPONENT_MAPPING;
189 srvDesc.Buffer.FirstElement = offset;
190 srvDesc.Buffer.NumElements = elementNum;
191 srvDesc.Buffer.StructureByteStride = (UINT)sizeof(StructBuffType);
192 srvDesc.Buffer.Flags = D3D12_BUFFER_SRV_FLAG_NONE;
193
194 // Create shader resource views
195 ID3D12Device* device = CApplication::GetAny().GetDXDevice();
196 SbDataPerComponent<StructBuffType>& buffer = m_buffers[*handle];
197 for (UINT i = 0; i < SCREEN_BUFFERING_NUM; ++i) {
198 device->CreateShaderResourceView(
199 buffer.structuredBuffers[i].GetResource(), &srvDesc,
200 buffer.descriptorHeaps[i].GetCpuHandle(meshIndex));
201 }
202}

206void ACSbAllocater<StructBuffType>::Release(MeshKey meshKey) {
207 auto it = m_dataPerMesh.find(meshKey);
208 if (it != m_dataPerMesh.end()) {
209 // Release per component data
210 if (it->second.handle) {
211 m_buffers.Erase(*it->second.handle);
212 }
213 // Release per mesh data
214 m_dataPerMesh.erase(it);
215 }

225// Check for key state
226template <class StructBuffType>
227inline SbAllocatorKeyState ACSbAllocater<StructBuffType>::CheckKey(MeshKey key) {
228 // Check if the key exists in the map
229 auto it = m_dataPerMesh.find(key);
230 if (it == m_dataPerMesh.end()) {
231 return SbAllocatorKeyState::NotExist;
232 }
233 // Check if key's data has not been calculated
234 else if (it->second.lastUpdateFrame != m_currentFrameNumber) {
235 return SbAllocatorKeyState::NotCalculated;
236 }
237 return SbAllocatorKeyState::Calculated;
238}
