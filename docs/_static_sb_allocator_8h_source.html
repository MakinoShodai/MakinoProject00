118// Copy data directly
119template<class StructBuffType>
120Utl::Dx::CPU_DESCRIPTOR_HANDLE ACStaticSbAllocator<StructBuffType>::DirectDataCopy(UINT elementsNum, const StructBuffType* src) {
121 // Get next back buffer index
122 UINT bufferIndex = CSwapChain::GetMain().GetNextBackBufferIndex();
123
124 if (elementsNum == 0) {
125 return m_descriptorHeap.GetCpuHandle(bufferIndex);
126 }
127
128 if (elementsNum != m_currentElementsNum[bufferIndex]) {
129 // Recreate structrued buffer
130 RecreateStructuredBuffer(bufferIndex, elementsNum);
131 }
132
133 // Perform map operation
134 m_structuredBuffers[bufferIndex]->Map();
135
136 // Update data of the structured buffer
137 m_structuredBuffers[bufferIndex]->DataCopy(src, elementsNum, 0);
138
139 // Get specified CPU handle from descriptor heap
140 return m_descriptorHeap.GetCpuHandle(bufferIndex);
141}

143// Recreate structured buffer
144template<class StructBuffType>
145void ACStaticSbAllocator<StructBuffType>::RecreateStructuredBuffer(UINT bufferIndex, UINT elementsNum) {
146 // Recreate structured buffer
147 m_structuredBuffers[bufferIndex].Reset(new CStructuredBufferResource<StructBuffType>());
148 m_structuredBuffers[bufferIndex]->Create(elementsNum);
149
150 // Set the number of elements
151 m_currentElementsNum[bufferIndex] = elementsNum;
152
153 // Create shader resource view desc
154 D3D12_SHADER_RESOURCE_VIEW_DESC srvDesc = {};
155 srvDesc.Format = DXGI_FORMAT_UNKNOWN;
156 srvDesc.ViewDimension = D3D12_SRV_DIMENSION_BUFFER;
157 srvDesc.Shader4ComponentMapping = D3D12_DEFAULT_SHADER_4_COMPONENT_MAPPING;
158 srvDesc.Buffer.FirstElement = 0;
159 srvDesc.Buffer.NumElements = elementsNum;
160 srvDesc.Buffer.StructureByteStride = (UINT)sizeof(StructBuffType);
161 srvDesc.Buffer.Flags = D3D12_BUFFER_SRV_FLAG_NONE;
162
163 // Recreate shader resource view
164 CApplication::GetAny().GetDXDevice()->CreateShaderResourceView(
165 m_structuredBuffers[bufferIndex]->GetResource(), &srvDesc,
166 m_descriptorHeap.GetCpuHandle(bufferIndex)
167 );
168}
