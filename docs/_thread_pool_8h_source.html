83// Dequeue one task to the queue
84template<class F, class ...Args>
85auto CThreadPool::CThreadSafeFeature::EnqueueTask(F&& function, Args && ...args) {
86 // Get the return value type
87 using ReturnType = decltype(function(std::forward<Args>(args)...));
88
89 // Create the wrapper function
90 // #NOTE : packaged_task can only supplement exception with the get function of future,
91 // so perform exception supplement processing before packaging
92 auto wrapperFunction = [function = std::forward<F>(function),
93 ...args = std::forward<Args>(args)]() -> ReturnType {
94 try {
95 return function(args...);
96 }
97 catch (const Utl::Error::CFatalError& e) {
98 // Error processing
99 CApplicationError::GetAny().Exit(e.WhatW());
100 throw e;
101 }
102 };
103
104 // Make the packaged_task object
105 auto task = std::make_shared<std::packaged_task<ReturnType()>>(wrapperFunction);
106
107 // Get the future object from the packaged_task object
108 std::future<ReturnType> fet = task->get_future();
109
110 // Operation on the queue
111 {
112 // Exclusion control for the queue
113 std::unique_lock<std::mutex> lock(m_owner->m_queueMutex);
114
115 // Enqueue task
116 m_owner->m_tasksQueue.emplace([task]() { (*task)(); });
117 }
118
119 // wakes up one of the waiting threads
120 m_owner->m_condition.notify_one();
121
122 // return the future object
123 return fet;
124}
