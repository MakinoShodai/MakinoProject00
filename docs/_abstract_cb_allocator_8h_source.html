106// Copy data directly
107template<class ConstBuffType>
108Utl::Dx::CPU_DESCRIPTOR_HANDLE ACCbAllocator<ConstBuffType>::DirectDataCopy(const ConstBuffType* src) {
109 // Get next back buffer index
110 UINT bufferIndex = CSwapChain::GetMain().GetNextBackBufferIndex();
111
112 // If the current descriptor heap limit is exceeded, advance the index in dynamic array
113 if (m_currentOffset >= m_numPerElement) {
114 m_currentOffset = 0;
115 m_currentDynamicArrayIndex++;
116
117 // Check for overflow of constant buffer size
118 if (m_currentDynamicArrayIndex >= m_constantBuffer[bufferIndex].size()) {
119 AddElement(bufferIndex);
120 }
121 }
122
123 // Perform map operation
124 m_constantBuffer[bufferIndex][m_currentDynamicArrayIndex].Map();
125
126 // Update data of the constant buffer
127 m_constantBuffer[bufferIndex][m_currentDynamicArrayIndex].DataCopy(src, m_currentOffset);
128
129 // Get specified CPU handle from descriptor heap and increment a current offset
130 return m_descriptorHeap[bufferIndex][m_currentDynamicArrayIndex].GetCpuHandle(m_currentOffset++);
131}

134template<class ConstBuffType>
135void ACCbAllocator<ConstBuffType>::RefreshCbAllocator() {
136 // Unmap
137 for (UINT i = 0; i <= m_currentDynamicArrayIndex; ++i) {
138 m_constantBuffer[CSwapChain::GetMain().GetNextBackBufferIndex()][i].Unmap();
139 }
140
141 // ClearData
142 m_currentOffset = 0;
143 m_currentDynamicArrayIndex = 0;
144}

146// Get handle for one previous allocated data
147template<class ConstBuffType>
148Utl::Dx::CPU_DESCRIPTOR_HANDLE ACCbAllocator<ConstBuffType>::GetPrevAllocatedData() {
149 // Get next back buffer index
150 UINT bufferIndex = CSwapChain::GetMain().GetNextBackBufferIndex();
151
152 // If the index in the dynamic array has just been updated, retrieve using the previous index
153 if (m_currentOffset == 0) {
154 throw Utl::Error::CStopDrawingSceneError(L"Warning! Trying to get previous data in constant buffer without allocating data.\n");
155 }
156
157 // Get previous data
158 return m_descriptorHeap[bufferIndex][m_currentDynamicArrayIndex].GetCpuHandle(m_currentOffset - 1);
159}

161// Add elements of constant buffer and descriptor heap
162template<class ConstBuffType>
163void ACCbAllocator<ConstBuffType>::AddElement(UINT index) {
164 // Add constant buffer
165 m_constantBuffer[index].emplace_back(m_numPerElement);
166 // Add Descriptor heap
167 m_descriptorHeap[index].emplace_back(D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV, D3D12_DESCRIPTOR_HEAP_FLAG_NONE, m_numPerElement);
168
169 // Get trailing variable
170 CConstantBufferResource<ConstBuffType>& cbuf = m_constantBuffer[index].back();
171 CDescriptorHeap& descriptor = m_descriptorHeap[index].back();
172
173 // Create all constant buffer view
174 D3D12_CONSTANT_BUFFER_VIEW_DESC cbvDesc = {};
175 cbvDesc.SizeInBytes = cbuf.GetSize();
176 for (UINT i = 0; i < m_numPerElement; ++i) {
177 // Get GPU address of constant buffer
178 cbvDesc.BufferLocation = cbuf.GetGPUVirtualAddress(i);
179
180 // Create constant buffer view
181 CApplication::GetAny().GetDXDevice()->CreateConstantBufferView(&cbvDesc, descriptor.GetCpuHandle(i));
182 }
183}
