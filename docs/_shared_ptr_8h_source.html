258template<class T>
259void CSharedPtr<T>::Reset(InstanceType* instance) {
260 if (m_strongRefCount != nullptr) {
261 // If strong reference count changes to 0 now, Release instance and set nullptr for weak reference
262 if (m_strongRefCount->m_refCount <= 1) {
263 if constexpr (std::is_array_v<T>) {
264 delete[] m_strongRefCount->m_ref;
265 }
266 else {
267 delete m_strongRefCount->m_ref;
268 }
269
270 if (m_weakRefCount) {
271 m_weakRefCount->m_ref = nullptr;
272 }
273 }
274
275 // Decrement strong reference count
276 RefCount<T>::Decrement(m_strongRefCount);
277 m_strongRefCount = nullptr;
278
279 // Decrement weak reference count
280 if (m_weakRefCount) {
281 RefCount<T>::Decrement(m_weakRefCount);
282 m_weakRefCount = nullptr;
283 }
284 }
285
286 // If it gains ownership of a new instance, it also recreates the reference count structure
287 if (instance != nullptr) {
288 m_strongRefCount = new RefCount<T>(instance);
289 m_weakRefCount = new RefCount<T>(instance);
290
291 // If the class creates a weak pointer with confidence, pass a reference count variable
292 if constexpr (std::is_base_of<IWeakPtrFromThis, T>::value) {
