165// Set sent graphics components to command list
166template<class T>
167void CGraphicsPipelineState::SetCommandForComponent(const std::vector<T*>& componentsArray, ID3D12Device* device, ID3D12GraphicsCommandList* cmdList, CDescriptorHeapPool* pool) {
168 for (T* component : componentsArray) {
169 if (!component->IsActiveOverall() || !component->IsDrawCondition()) { continue; }
170
171 // Draw on all meshes the component has
172 UINT meshNum = component->GetMeshNum();
173 for (UINT meshIndex = 0; meshIndex < meshNum; ++meshIndex) {
174 // Get the mesh's vertex buffer and index buffer
175 const CMeshBuffer* meshBuff = component->GetMeshBuffer(meshIndex);
176 const CVertexBuffer* vertBuff = meshBuff->GetVertex();
177 const CIndexBuffer* indexBuff = meshBuff->GetIndex();
178 // If can't find either of them, output a message
179 if (vertBuff == nullptr || indexBuff == nullptr) {
180 OutputDebugString(L"Warning! No vertex or index buffer is allocated for the graphics component\n");
181 continue;
182 }
183
184 // Checks if the descriptor heap region overflows, and if so, advances to the next descriptor heap
185 CheckRegionOverflow();
186
187 // Create constant buffer views, that need to be updated on a per-object basis, for component
188 for (auto& cbv : m_dynamicAllocators) {
189 device->CopyDescriptorsSimple(1,
190 pool->GetCPUHandle(cbv.second),
191 cbv.first->GetAllocatedDataHandle(component, meshIndex), D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV);
192 }
193
194 // Create dynamic textures for component
195 if (m_totalDynamicTextureNum > 0) {
196 // Ask for textures assignment information to be built
197 bool isAllAllocate = component->AskToAllocateTextures(meshIndex, *this, m_totalDynamicTextureNum);
198
199 // If the necessary texture for this GPSO is not found, output a message
200 if (!isAllAllocate) {
201 throw Utl::Error::CStopDrawingSceneError(L"Warning! The necessary texture for this GPSO is not found\n");
202 }
203 }
204
205 // Draw object
206 {
207 // Associate descriptor heap with root parameter
208 UINT staticTableSize = (UINT)m_staticRootDescriptorTableOffset.size();
209 UINT tableSize = (UINT)m_dynamicRootDescriptorTableOffset.size();
210 for (UINT i = 0; i < tableSize; ++i) {
211 cmdList->SetGraphicsRootDescriptorTable(staticTableSize + i,
212 pool->GetGPUHandle(m_dynamicRootDescriptorTableOffset[i]));
213 }
214
215 // Set vertex buffer view
216 cmdList->IASetVertexBuffers(0, 1, vertBuff->GetView());
217 // Set index buffer view
218 cmdList->IASetIndexBuffer(indexBuff->GetView());
219
220 // Set primitive topology
221 cmdList->IASetPrimitiveTopology(meshBuff->GetTopologyType());
222
223 // Draw
224 cmdList->DrawIndexedInstanced(
225 indexBuff->GetNumIndices(), // The number of indices
226 1, // The number of instance
227 0, // Index offset
228 0, // Vertex offset
229 0 // Instance offset
230 );
231 }
232
233 // Determine the use of the region based on the number of dynamic resources per object
234 pool->DetermineUseRegion(m_totalDynamicDescRangeNum);
235 }
236 }
237}
