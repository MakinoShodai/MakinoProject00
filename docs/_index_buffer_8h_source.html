75// Create index buffer
76template<class TIndex>
77 requires IsIndexTypeUINT<TIndex>::value
78void CIndexBuffer::Create(const void* indicesData, UINT numIndices) {
79 // Copy index information
80 m_numIndices = numIndices;
81 m_bytesPerIndex = sizeof(TIndex);
82
83 // Calculate bytes of all indices
84 UINT64 bytesAllIndices = static_cast<UINT64>(m_numIndices) * m_bytesPerIndex;
85
86 // Define mapping function
87 auto mappingFunc = [&](UINT8* mappedBuffer) {
88 memcpy(mappedBuffer, indicesData, bytesAllIndices);
89 };
90
91 // Create resource
92 CreateResource(bytesAllIndices, D3D12_RESOURCE_STATE_INDEX_BUFFER, mappingFunc);
93
94 // Create index buffer view
95 m_ibView.BufferLocation = m_resource->GetGPUVirtualAddress();
96 m_ibView.Format = GetIndexFormat<TIndex>();
97 m_ibView.SizeInBytes = (UINT)bytesAllIndices;
98}
