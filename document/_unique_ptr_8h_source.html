200template<class T>
201void CUniquePtrWeakable<T>::Reset(InstanceType* instance) {
202 // Delete instance
203 if (m_instance != nullptr) {
204 if constexpr (std::is_array_v<T>) {
205 delete[] m_instance;
206 }
207 else {
208 delete m_instance;
209 }
210
211 // Decrement references to instances by replacing them with nullptr
212 if (m_refCount) {
213 m_refCount->m_ref = nullptr;
214 RefCount<T>::Decrement(m_refCount);
215 m_refCount = nullptr;
216 }
217 }
218
219 // If it gains ownership of a new instance, it also recreates the reference count structure
220 m_instance = instance;
221 if (m_instance != nullptr) {
222 m_refCount = new RefCount<T>(m_instance);
223
224 // If the class creates a weak pointer with confidence, pass a reference count variable
225 if constexpr (std::is_base_of<IWeakPtrFromThis, T>::value) {
