128// Internal function of 'GetAllocatedDataHandle'
129template<class StructBuffType>
130template<class T>
131 requires std::is_base_of_v<ACGraphicsComponent, T>
132Utl::Dx::CPU_DESCRIPTOR_HANDLE ACDynamicSbAllocatorPerMesh<StructBuffType>::InnerGetAllocateDataHandle(T* component, UINT meshIndex) {
133 // Check the state of the mesh of the sent component
134 MeshKey sentComponentKey = component->GetMeshKey(meshIndex);
135 SbAllocatorKeyState flag = ACSbAllocater<StructBuffType>::CheckKey(sentComponentKey);
136 switch (flag) {
137 // If the key doesn't exist in the map, create new structured buffer
138 case SbAllocatorKeyState::NotExist:
139 this->CreateBuffer(component);
140 break;
141 // If key's data has been calculated, return its handle
142 case SbAllocatorKeyState::Calculated:
143 return ACSbAllocater<StructBuffType>::GetCPUHandle(sentComponentKey);
144 break;
145 default:
146 break;
147 }
148
149 // Calculate structured buffer data, and return its handle
150 return this->AllocateData(component, meshIndex);
151}
