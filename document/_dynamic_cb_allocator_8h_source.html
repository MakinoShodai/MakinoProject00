199// Internal function of 'GetAllocatedDataHandle'
200template<class ConstBuffType>
201template<class T>
202 requires std::is_base_of_v<ACGraphicsComponent, T>
203Utl::Dx::CPU_DESCRIPTOR_HANDLE ACDynamicCbAllocatorPerMesh<ConstBuffType>::InnerGetAllocateDataHandle(T* component, UINT meshIndex) {
204 // If the cache already has calculated data, return it
205 MeshKey sentComponentKey = component->GetMeshKey(meshIndex);
206 auto it = m_cachedHandleMap.find(sentComponentKey);
207 if (it != m_cachedHandleMap.end()) {
208 return m_cachedHandleMap[sentComponentKey];
209 }
210
211 // Calculate constant buffer data, and get its handle
212 Utl::Dx::CPU_DESCRIPTOR_HANDLE handle = AllocateData(component, meshIndex);
213
214 // Set the handle to the cache, and return it
215 m_cachedHandleMap.emplace(sentComponentKey, handle);
216 return handle;
217}

223Utl::Dx::CPU_DESCRIPTOR_HANDLE ACDynamicCbAllocator<ConstBuffType>::InnerGetAllocateDataHandle(T* component, UINT meshIndex) {
224 // #NOTE : Don't separate the process for each mesh held by the sent component

226 MeshKey sentComponentKey = component->GetMeshKey(0);

230 }

234

236 ACDynamicCbAllocatorPerMesh<ConstBuffType>::m_cachedHandleMap.emplace(sentComponentKey, handle);
