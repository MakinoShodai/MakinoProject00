112// Generate ID
113template <class Resource>
114UniqueID CIDResourceFactory<Resource>::GenerateID() {
115 UniqueID uniqueID;
116 // If a released IDs exists, use it
117 if (!m_freeID.empty()) {
118 // Get the oldest ID
119 uniqueID = m_freeID.front();
120 m_freeID.pop();
121
122 // Extract the version part from the got ID and add the version
123 UniqueID_Version version = static_cast<UniqueID_Version>(uniqueID >> UNIQUE_ID_INDEX_BITNUM);
124 version++;
125
126 // Merge the version part and the index part
127 uniqueID = (static_cast<UniqueID>(version) << UNIQUE_ID_INDEX_BITNUM) | (uniqueID & UNIQUE_ID_INDEX_BITMASK);
128 }
129 else {
130 // Assign the next unique ID
131 uniqueID = static_cast<UniqueID>(this->m_nextIndex);
132
133 // If the maximum index value is exceeded, an exception is handled
134 if (this->m_nextIndex >= UNIQUE_ID_INDEX_BITMASK) {
135 throw std::overflow_error("The index of the IDResource class has exceeded the limit of the index part.");
136 }
137
138 // Extend array by one
139 this->m_resourceArray.emplace_back(nullptr);
140
141 // Add the next index
142 this->m_nextIndex++;
143 }
144
145 return uniqueID;
146}
