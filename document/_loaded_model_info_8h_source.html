588// Compute current time value from keyframes
589template<typename T, typename LerpFunc>
590T CDynamicModelController::ComputeValueFromKeyFrame(const std::vector<ModelInfo::KeyFrame<T>>& keyFrames, float currentTime, const LerpFunc& lerpFunc) const {
591 // If the current time is before the first key, the value of the first key are used
592 if (currentTime <= keyFrames[0].time) {
593 return keyFrames[0].value;
594 }
595 // If the current time is after the last key, the value of the last key are used
596 else if (keyFrames.back().time <= currentTime) {
597 return keyFrames.back().value;
598 }
599 // If not, compute current time value
600 else {
601 UINT size = (UINT)keyFrames.size() - 1;
602 for (UINT i = 0; i < size; ++i) {
603 if (keyFrames[i].time < currentTime && currentTime <= keyFrames[i + 1].time) {
604 float timeLen = keyFrames[i + 1].time - keyFrames[i].time;
605 float rate = (currentTime - keyFrames[i].time) / timeLen;
606 return lerpFunc(keyFrames[i].value, keyFrames[i + 1].value, rate);
607 }
608 }
609 }
610 // Return the last value in case of a problem
611 return keyFrames.back().value;
612}
