557// Calculate the angle of rotation from this vector to another vector in 2D, ignoring one axis
558template<Utl::Type::Traits::IsFloatingPoint T, uint8_t SIZE>
559float Vector<T, SIZE>::AngleTo(const Vector& v, uint8_t ignoreAxis) const requires(SIZE == 3) {
560 // Calculate dot product and cross product
561 T d = Utl::Clamp(Dot(v), T(-1), T(1));
562
563 T epsilon = std::numeric_limits<T>::epsilon();
564 // Vectors are parallel
565 if (std::abs(d - T(1)) < epsilon) {
566 return T(0);
567 }
568 // Vectors are antiparallel
569 else if (std::abs(d + T(1)) < epsilon) {
570 return Utl::GetPI<T>();
571 }
572 // Calculate angle normally
573 else {
574 // Convert [0, Pi] to [-Pi, Pi]
575 Vector c = Cross(v);
576 if (c[ignoreAxis] < T(0))
577 return -std::acos(d);
578 else
579 return std::acos(d);
580 }
581}

583// Get a inverse matrix
584template<Utl::Type::Traits::IsFloatingPoint T, uint8_t ROW, uint8_t COLUMN>
585Matrix<T, ROW, COLUMN> Matrix<T, ROW, COLUMN>::Inverse() {
586 static_assert(ROW == COLUMN, "Inverse only defined for square matrix.");
587
588 // Create own copy matrix and unit matrix
589 Matrix copy = *this;
590 Matrix inv = Identity();
591
592 T tmp;
593 for (uint8_t i = 0; i < ROW; ++i) {
594 // Partial Pivot Selection
595 // Bring the element in i column, with the highest absolute value to the diagonal component
596 // #NOTE : Since the operation is finished up to the i - 1 row,
597 // the identity matrix is collapsed if the rows are swapped.
598 T maxAbs = std::abs(copy(i, i));
599 for (uint8_t row = i; row < ROW; ++row) {
600 // If the absolute value of it is highest than maxAbs
601 tmp = std::abs(copy(row, i));
602 if (maxAbs < tmp) {
603 maxAbs = tmp;
604 // Replace all elements in one row
605 for (uint8_t col = 0; col < COLUMN; ++col) {
606 tmp = copy(i, col);
607 copy(i, col) = copy(row, col);
608 copy(row, col) = tmp;
609
610 // Perform the same operation
611 tmp = inv(i, col);
612 inv(i, col) = inv(row, col);
613 inv(row, col) = tmp;
614 }
615 }
616 }
617
618 // Perform normalization operation on all columns of i row
619 T invDiagonal = T(1) / copy(i, i);
620 // #NOTE : Operation is not needed because the elements are guaranteed to be 0 up to the i - 1 column of i row.
621 for (uint8_t col = i; col < COLUMN; ++col) {
622 copy(i, col) *= invDiagonal;
623 }
624 // Perform the same operation
625 // #NOTE : This matrix does not aim to be a identity matrix, so it is not guaranteed to be 0 up to the i - 1 column in i row.
626 for (uint8_t col = 0; col < COLUMN; ++col) {
627 inv(i, col) *= invDiagonal;
628 }
629
630 // Calculate to 0 for all elements in a column except for the diagonal component
631 for (uint8_t row = 0; row < ROW; ++row) {
632 if (row != i) {
633 T r = -copy(row, i);
634
635 // Calculate this element to zero
636 // #NOTE : Operation is not needed because the elements are guaranteed to be zero up to the 'i - 1' column of i row.
637 for (uint8_t col = i; col < COLUMN; ++col) {
638 copy(row, col) += copy(i, col) * r;
639 }
640 // Perform the same operation
641 // #NOTE : This matrix does not aim to be a identity matrix, so it is not guaranteed to be 0 up to the i - 1 column in i row.
642 for (uint8_t col = 0; col < COLUMN; ++col) {
643 inv(row, col) += inv(i, col) * r;
644 }
645 }
646 }
647 }
648
649 return inv;
650}

666template<Utl::Type::Traits::IsFloatingPoint T>
667Quaternion<T>::Quaternion(const Vector<T, 3>& eulerAngle) {
668 // Prepare variables needed for calculations
669 T cosX = std::cos(eulerAngle.x() * Utl::Inv::Get2<T>());
670 T sinX = std::sin(eulerAngle.x() * Utl::Inv::Get2<T>());
671 T cosY = std::cos(eulerAngle.y() * Utl::Inv::Get2<T>());
672 T sinY = std::sin(eulerAngle.y() * Utl::Inv::Get2<T>());
673 T cosZ = std::cos(eulerAngle.z() * Utl::Inv::Get2<T>());
674 T sinZ = std::sin(eulerAngle.z() * Utl::Inv::Get2<T>());
675 // Calculate each components
676 x() = sinX * cosY * cosZ - cosX * sinY * sinZ;
677 y() = cosX * sinY * cosZ + sinX * cosY * sinZ;
678 z() = cosX * cosY * sinZ - sinX * sinY * cosZ;
679 w() = cosX * cosY * cosZ + sinX * sinY * sinZ;
680}

682// Get Eular angle from this quaternion
683template<Utl::Type::Traits::IsFloatingPoint T>
684const Vector<T, 3> Quaternion<T>::GetEularAngle() const {
685 constexpr T MINUSONE_EPSILON = T(-1) + FLT_EPSILON;
686 constexpr T PLUSONE_EPSILON = T(1) - FLT_EPSILON;
687
688 // Prepare variables needed for calculations
689 T wy = w() * y();
690 T xz = x() * z();
691
692 // Calculate the value to be put into asin
693 T forAsin = T(-2) * (xz - wy);
694
695 // If the absolute value of this value is greater than 1,
696 // asin returns an error value and performs a special calculation
697 // #NOTE : In this case, gimbal lock has occurred, so rotation can only be expressed as the total angle of roll and yaw.
698 // Set x to 0 and assign the total angle to z
699
700 // If it is negative
701 if (forAsin <= MINUSONE_EPSILON) {
702 return Vector<T, 3>(
703 T(0),
704 -Utl::GetPI<T>() * T(0.5),
705 T(2) * std::atan2(x(), -y())
706 );
707 }
708 // If it is positive
709 else if (forAsin >= PLUSONE_EPSILON) {
710 return Vector<T, 3>(
711 T(0),
712 Utl::GetPI<T>() * T(0.5),
713 T(2) * std::atan2(-x(), y())
714 );
715 }
716 else {
717 // Prepare the variables needed for the remaining calculations
718 T ww = w() * w();
719 T wx = w() * x();
720 T wz = w() * z();
721 T xx = x() * x();
722 T xy = x() * y();
723 T yy = y() * y();
724 T yz = y() * z();
725 T zz = z() * z();
726 // Convert and return
727 return Vector<T, 3>(
728 std::atan2(T(2) * (yz + wx), ww - xx - yy + zz), // Roll
729 std::asin(forAsin), // Pitch
730 std::atan2(T(2) * (xy + wz), ww + xx - yy - zz) // Yaw
731 );
732 }
733}

735// Get rotation matrix from this quaternion
736template<Utl::Type::Traits::IsFloatingPoint T>
737const Matrix<T, 3, 3> Quaternion<T>::GetMatrix() const {
738 Matrix<T, 3, 3> mat;
739 // Prepare variables needed for calculations
740 T xx = x() * x();
741 T yy = y() * y();
742 T zz = z() * z();
743 T xy = x() * y();
744 T xz = x() * z();
745 T yz = y() * z();
746 T wx = w() * x();
747 T wy = w() * y();
748 T wz = w() * z();
749 // Calculate each components
750 mat(0, 0) = T(1) - T(2) * (yy + zz);
751 mat(0, 1) = T(2) * (xy - wz);
752 mat(0, 2) = T(2) * (xz + wy);
753 mat(1, 0) = T(2) * (xy + wz);
754 mat(1, 1) = T(1) - T(2) * (xx + zz);
755 mat(1, 2) = T(2) * (yz - wx);
756 mat(2, 0) = T(2) * (xz - wy);
757 mat(2, 1) = T(2) * (yz + wx);
758 mat(2, 2) = T(1) - T(2) * (xx + yy);
759 return mat;
760}

762// Get angular velocity from this quaternion
763template<Utl::Type::Traits::IsFloatingPoint T>
764Vector<T, 3> Quaternion<T>::GetAngularVelocity() const {
765 // If the denominator is zero, the angular velocity is zero
766 T denom = std::sqrt(T(1) - w() * w());
767 if (std::abs(denom) <= FLT_EPSILON) { return Vector<T, 3>(0.0f, 0.0f, 0.0f); }
768
769 // Prepare variables needed for calculations
770 T angle = T(2) * std::acos(w());
771 T invSinHalfAngle = T(1) / denom;
772 // Calculate each components
773 return Vector<T, 3>(
774 angle * x() * invSinHalfAngle,
775 angle * y() * invSinHalfAngle,
776 angle * z() * invSinHalfAngle
777 );
778}

796// Apply this quaternion rotation to a 3 dimensional vector
797template<Utl::Type::Traits::IsFloatingPoint T>
798const Vector<T, 3> Quaternion<T>::operator*(const Vector<T, 3>& v) const {
799 // Replace 3D vector with quaternion
800 Quaternion v2(v.x(), v.y(), v.z(), T(0));
801 // Apply this quaternion rotation to replaced 3D vector
802 // #NOTE : When the conjugate quaternion is p, 'q * v * p'
803 Quaternion vRotate = Product(Product(*this, v2), GetConjugation());
804 // Extract x,y,z components
805 return Vector<T, 3>(vRotate.x(), vRotate.y(), vRotate.z());
806}

808// Quaternion product
809template<Utl::Type::Traits::IsFloatingPoint T>
810inline Quaternion<T> Quaternion<T>::Product(const Quaternion& lhs, const Quaternion& rhs) {
811 return Quaternion(
812 lhs.w() * rhs.x() - lhs.z() * rhs.y() + lhs.y() * rhs.z() + lhs.x() * rhs.w(),
813 lhs.z() * rhs.x() + lhs.w() * rhs.y() - lhs.x() * rhs.z() + lhs.y() * rhs.w(),
814 -lhs.y() * rhs.x() + lhs.x() * rhs.y() + lhs.w() * rhs.z() + lhs.z() * rhs.w(),
815 -lhs.x() * rhs.x() - lhs.y() * rhs.y() - lhs.z() * rhs.z() + lhs.w() * rhs.w()
816 );
817}
